from google.colab import files import pandas as pd import io import json import itertools from collections import defaultdict from datetime import datetime, timedelta # --- 1. エクセルファイルのアップロード --- print("エクセルファイル（.xlsx）を選択してください:") uploaded = files.upload() file_name = list(uploaded.keys())[0] df = pd.read_excel(io.BytesIO(uploaded[file_name])) # --- 2. データ整理 --- df.columns = df.columns.str.strip() exclude_keywords = ['退職', '再任用', '期限付', '割愛', '教育委員会', '教委', 'センター', '研修', '事務局'] # 異動日をdatetime型に変換（無い場合は年度の4月1日とする） def parse_date(row): if pd.notna(row.get('異動日')): try: return pd.to_datetime(row['異動日']) except: pass try: return pd.to_datetime(f"{int(row['年度'])}-04-01") except: return None df['parsed_date'] = df.apply(parse_date, axis=1) df = df.dropna(subset=['parsed_date']) # 名簿 unique_names = df['氏名'].dropna().unique() name_to_id = {name: int(i + 1000) for i, name in enumerate(unique_names)} teacher_db = [{"id": v, "name": k} for k, v in name_to_id.items()] # --- 3. イベントベースでの経歴（Tenures）の再構築 --- events_by_teacher = defaultdict(list) for _, row in df.iterrows(): t_id = name_to_id[row['氏名']] date = row['parsed_date'] role = str(row['役職']).strip() if pd.notna(row['役職']) else "" # 前所属（その学校から「離任」したイベント） # ※ 離任日は異動日の「前日（3月31日など）」とする prev_school = str(row['前所属']).strip() if pd.notna(row['前所属']) and prev_school != 'nan' and not any(k in prev_school for k in exclude_keywords): leave_date = date - timedelta(days=1) events_by_teacher[t_id].append({ 'school': prev_school, 'type': 'leave', 'date': leave_date, 'role': role }) # 新所属（その学校へ「着任」したイベント） curr_school = str(row['新所属']).strip() if pd.notna(row['新所属']) and curr_school != 'nan' and not any(k in curr_school for k in exclude_keywords): events_by_teacher[t_id].append({ 'school': curr_school, 'type': 'join', 'date': date, 'role': role }) tenures = [] for t_id, events in events_by_teacher.items(): # 学校ごとにイベントを分類し、日付順にソート school_events = defaultdict(list) for e in events: school_events[e['school']].append(e) for school, evs in school_events.items(): evs.sort(key=lambda x: x['date']) i = 0 while i < len(evs): e = evs[i] if e['type'] == 'join': start_date = e['date'] inf_start = False # 次のイベントが同じ学校の離任なら「確定期間」 if i + 1 < len(evs) and evs[i+1]['type'] == 'leave': end_date = evs[i+1]['date'] inf_end = False i += 2 # 2つ分のイベントを処理した else: # 離任記録がないので未来へ6年（約2190日）推定 end_date = start_date + pd.DateOffset(years=6) inf_end = True i += 1 else: # e['type'] == 'leave' # 着任記録がないまま離任記録だけがある場合（過去へ6年推定） end_date = e['date'] start_date = end_date - pd.DateOffset(years=6) inf_start = True inf_end = False i += 1 tenures.append({ "teacher_id": t_id, "school": school, "start_date": start_date.strftime('%Y-%m-%d'), "end_date": end_date.strftime('%Y-%m-%d'), "start_year": start_date.year, # HTML互換性のため残す "end_year": end_date.year, "inferred_start": inf_start, "inferred_end": inf_end }) # --- 4. 日付ベースでの同僚リンク計算 --- colleague_links = [] school_groups = defaultdict(list) for t in tenures: school_groups[t['school']].append(t) seen = set() for school, records in school_groups.items(): for t1, t2 in itertools.combinations(records, 2): if t1['teacher_id'] == t2['teacher_id']: continue # 文字列の日付を比較して重なりをチェック os_date = max(t1['start_date'], t2['start_date']) oe_date = min(t1['end_date'], t2['end_date']) if os_date <= oe_date: pair = tuple(sorted((t1['teacher_id'], t2['teacher_id']))) if (pair, school, os_date, oe_date) in seen: continue # 推定境界の判定 is_inf = (os_date == t1['start_date'] and t1['inferred_start']) or \ (os_date == t2['start_date'] and t2['inferred_start']) or \ (oe_date == t1['end_date'] and t1['inferred_end']) or \ (oe_date == t2['end_date'] and t2['inferred_end']) # ラベル用に年だけ抽出 os_year = os_date.split('-')[0] oe_year = oe_date.split('-')[0] colleague_links.append({ "from": t1['teacher_id'], "to": t2['teacher_id'], "school": school, "label": f"{os_year}-{oe_year}" + (" [推定]" if is_inf else ""), "is_inferred": is_inf, "period": [int(os_year), int(oe_year)], "overlap_start": os_date, "overlap_end": oe_date }) seen.add((pair, school, os_date, oe_date)) # --- 5. ファイル保存（軽量化） --- files_to_save = { 'teachers_db.json': teacher_db, 'teachers_tenures.json': tenures, 'colleague_links.json': colleague_links } for name, data in files_to_save.items(): with open(name, 'w', encoding='utf-8') as f: json.dump(data, f, ensure_ascii=False) files.download(name) print("✅ 新しいロジックでJSONファイルを生成しました！")
